\UASectionSlide{Motivación}

\begin{frame}{Del estado al flujo}
\begin{itemize}
\item Antes: estado persistente (clase 6)
\item Ahora: \textbf{flujo continuo de eventos}
\end{itemize}

\pause
\begin{block}{Problema}
¿Cómo procesar datos en tiempo real sin perder orden ni consistencia?
\end{block}
\end{frame}

\begin{frame}{Idea central}
\begin{block}{}
El sistema es un \textbf{log inmutable de eventos}
\end{block}

\pause
\begin{itemize}
\item append-only
\item ordenado
\item reproducible
\end{itemize}
\end{frame}

\UASectionSlide{Commit Log}

\begin{frame}{Definición}
\begin{block}{}
Secuencia ordenada e inmutable: $L = (e_0, e_1, ..., e_n)$
\end{block}

\pause
\begin{itemize}
\item cada evento tiene \textbf{offset}
\item lectura secuencial
\end{itemize}
\end{frame}

\begin{frame}{Offsets}
\begin{itemize}
\item índice lógico del evento
\item monotónico por partición
\item controlado por el consumidor
\end{itemize}
\end{frame}

\UASectionSlide{Kafka}

\begin{frame}{Arquitectura}
\begin{itemize}
\item Topic = categoría lógica
\item Partition = log ordenado
\item Broker = nodo de almacenamiento
\end{itemize}
\end{frame}

\begin{frame}{Particiones y orden}
\begin{center}
\begin{tikzpicture}[node distance=1.5cm]
\node[draw] (p0) {P0: e0,e1,e2};
\node[draw,below of=p0] (p1) {P1: e0,e1};

\end{tikzpicture}
\end{center}

\pause
\begin{alertblock}{Clave}
El orden es \textbf{local a la partición}, no global
\end{alertblock}
\end{frame}

\begin{frame}{Replicación}
\begin{itemize}
\item líder (leader)
\item seguidores (followers)
\item ISR (in-sync replicas)
\end{itemize}
\end{frame}

\UASectionSlide{Semánticas de entrega}

\begin{frame}{At-most-once}
\begin{itemize}
\item commit de offset antes de procesar
\item puede perder mensajes
\item nunca duplica
\end{itemize}
\end{frame}

\begin{frame}{At-least-once}
\begin{itemize}
\item commit después de procesar
\item puede duplicar
\item no pierde mensajes
\end{itemize}
\end{frame}

\begin{frame}{Exactly-once}
\begin{itemize}
\item no duplicación
\item no pérdida
\end{itemize}

\pause
\begin{alertblock}{Realidad}
Se logra combinando:
\begin{itemize}
\item idempotencia
\item transacciones
\item control de offsets
\end{itemize}
\end{alertblock}
\end{frame}
