\UASectionSlide{Motivación}

\begin{frame}{Del modelo lógico al físico}
Hasta ahora:
\begin{itemize}
\item transacciones
\item serializabilidad
\end{itemize}

\pause
\begin{block}{Problema}
¿Cómo implementamos esto sobre disco real?
\end{block}
\end{frame}

\begin{frame}{Restricciones físicas}
\begin{itemize}
\item escritura secuencial >> aleatoria
\item fsync es costoso
\item memoria limitada
\end{itemize}
\end{frame}

\UASectionSlide{Write-Ahead Logging}

\begin{frame}{Regla WAL}
\begin{block}{}
Nunca escribir datos persistentes antes de loguear la operación
\end{block}

\pause
\begin{itemize}
\item asegura atomicidad
\item permite recovery
\end{itemize}
\end{frame}

\begin{frame}{Ejemplo}
\begin{itemize}
\item T1: write(x=5)
\item WAL: “write x=5”
\item crash
\end{itemize}

\pause
Recovery:
\begin{itemize}
\item REDO ⇒ reaplicar
\end{itemize}
\end{frame}

\UASectionSlide{MVCC}

\begin{frame}{Idea}
\begin{itemize}
\item múltiples versiones por registro
\item cada transacción ve snapshot
\end{itemize}
\end{frame}

\begin{frame}{Lectura}
\begin{itemize}
\item seleccionar versión con timestamp ≤ snapshot
\end{itemize}
\end{frame}

\begin{frame}{Ventaja}
\begin{itemize}
\item lectores no bloquean escritores
\item alto paralelismo
\end{itemize}
\end{frame}

\UASectionSlide{LSM Trees}

\begin{frame}{Problema de discos}
\begin{itemize}
\item random write costoso
\end{itemize}

\pause
Solución:
\begin{block}{}
convertir escrituras en secuenciales
\end{block}
\end{frame}

\begin{frame}{Arquitectura}
\begin{itemize}
\item Memtable (RAM)
\item WAL
\item SSTables (disco)
\end{itemize}
\end{frame}

\begin{frame}{Flujo}
\begin{center}
\begin{tikzpicture}[node distance=2cm]
\node[draw] (wal) {WAL};
\node[draw, right of=wal] (mem) {Memtable};
\node[draw, below of=mem] (sst) {SSTable};

\draw[->] (wal) -- (mem);
\draw[->] (mem) -- (sst);
\end{tikzpicture}
\end{center}
\end{frame}

\begin{frame}{Compaction}
\begin{itemize}
\item merge de SSTables
\item elimina versiones viejas
\end{itemize}
\end{frame}

\UASectionSlide{Índices}

\begin{frame}{B-Tree}
\begin{itemize}
\item balanceado
\item acceso logarítmico
\end{itemize}
\end{frame}

\begin{frame}{Comparación}
\begin{itemize}
\item LSM: write-heavy
\item B-Tree: read-heavy
\end{itemize}
\end{frame}

\UASectionSlide{Cierre}

\begin{frame}{Idea central}
\begin{block}{}
El almacenamiento no es implementación: es diseño del sistema
\end{block}
\end{frame}
