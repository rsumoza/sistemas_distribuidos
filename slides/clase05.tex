\UASectionSlide{Tiempo y causalidad}

\begin{frame}{Objetivos de la clase}
\begin{itemize}
\item entender por qué el tiempo físico no alcanza para ordenar eventos distribuidos
\item formalizar la noción de causalidad
\item introducir relojes lógicos de Lamport y vector clocks
\item distinguir orden total, orden parcial y concurrencia
\end{itemize}
\end{frame}

\begin{frame}{Motivación}
En un sistema distribuido:
\begin{itemize}
\item cada nodo observa su reloj local
\item los relojes pueden derivar
\item la red introduce demora variable
\end{itemize}

\begin{block}{Pregunta}
Si dos eventos ocurrieron en nodos distintos, ¿podemos decir con certeza cuál ocurrió antes?
\end{block}
\end{frame}

\begin{frame}{Por qué el tiempo físico no alcanza}
Dos nodos pueden registrar timestamps consistentes localmente y, sin embargo:
\begin{itemize}
\item un mensaje puede llegar mucho después de haberse emitido
\item una corrección NTP puede modificar el reloj local
\item la comparación entre timestamps físicos no siempre preserva dependencia causal
\end{itemize}

\begin{uawarn}
Ordenar por tiempo físico no equivale a ordenar por dependencia causal.
\end{uawarn}
\end{frame}

\begin{frame}{Relación happened-before}
Definimos la relación causal $a \rightarrow b$:
\begin{enumerate}
\item si $a$ y $b$ ocurren en el mismo proceso y $a$ ocurre antes que $b$
\item si $a$ es el envío de un mensaje y $b$ es la recepción del mismo mensaje
\item por transitividad
\end{enumerate}

Si ni $a \rightarrow b$ ni $b \rightarrow a$, entonces los eventos son \textbf{concurrentes}.
\end{frame}

\end{frame}

\begin{frame}{Relojes de Lamport}
Cada proceso mantiene un contador lógico $L_i$.

Reglas:
\begin{itemize}
\item antes de cada evento local: $L_i \leftarrow L_i + 1$
\item al enviar un mensaje, se adjunta el valor actual
\item al recibir un mensaje con timestamp $t$: $L_i \leftarrow \max(L_i,t)+1$
\end{itemize}

Propiedad:
\[
a \rightarrow b \implies L(a) < L(b)
\]
\end{frame}

\begin{frame}{Límite de Lamport}
Lamport preserva causalidad, pero no detecta concurrencia.

Puede ocurrir:
\[
L(a) < L(b)
\]
sin que exista
\[
a \rightarrow b
\]

\begin{block}{Conclusión}
Lamport induce un orden compatible con la causalidad, pero no caracteriza completamente la relación causal.
\end{block}
\end{frame}

\begin{frame}{Vector clocks}
Cada proceso mantiene un vector $V_i[1..N]$.

Intuición:
\begin{itemize}
\item cada componente registra lo que el proceso sabe sobre cada proceso
\item al ocurrir un evento local, incrementa su propia componente
\item al recibir un mensaje, hace máximo componente a componente
\end{itemize}
\end{frame}

\begin{frame}{Propiedad de vector clocks}
Para eventos $a,b$:
\[
a \rightarrow b \iff V(a) < V(b)
\]
donde $V(a) < V(b)$ significa:
\begin{itemize}
\item $V(a)[k] \le V(b)[k]$ para todo $k$
\item y estrictamente menor en al menos una componente
\end{itemize}

Si los vectores no son comparables, los eventos son concurrentes.
\end{frame}

\begin{frame}{Ejemplo: detección de concurrencia}
\[
V(a)=(2,1,0), \qquad V(b)=(1,2,0)
\]

Ninguno domina al otro. Entonces:
\[
a \parallel b
\]

\begin{block}{Valor práctico}
Vector clocks permiten distinguir entre:
\begin{itemize}
\item actualización derivada de otra
\item actualización concurrente
\end{itemize}
\end{block}
\end{frame}

\begin{frame}{Aplicaciones reales}
\begin{itemize}
\item resolución de conflictos en stores eventualmente consistentes
\item versionado de objetos
\item depuración de sistemas distribuidos
\item reconstrucción causal de ejecuciones
\end{itemize}
\end{frame}

\begin{frame}{Ejemplo tipo MIT}
Supongamos dos réplicas desconectadas que aceptan escrituras concurrentes sobre el mismo objeto.

\begin{itemize}
\item con timestamps físicos: una puede sobrescribir arbitrariamente a la otra
\item con vector clocks: se detecta concurrencia y se fuerza reconciliación
\end{itemize}

\begin{block}{Pregunta}
¿Qué preferimos en un store distribuido real: simplicidad o preservación de causalidad?
\end{block}
\end{frame}

\begin{frame}{Errores frecuentes}
\begin{itemize}
\item creer que sincronizar relojes físicos resuelve el problema de orden
\item confundir orden total con causalidad
\item pensar que Lamport detecta concurrencia
\item olvidar que vector clocks crecen con el número de procesos
\end{itemize}
\end{frame}

\begin{frame}{Cierre}
\begin{block}{Idea clave}
El tiempo distribuido no se razona como reloj universal, sino como dependencia causal entre eventos.
\end{block}
\end{frame}
