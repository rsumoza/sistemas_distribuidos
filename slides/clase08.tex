\UASectionSlide{Motivación}

\begin{frame}{De modelos a sistemas reales}
Hasta ahora vimos:
\begin{itemize}
\item consistencia como contrato
\item replicación
\item consenso
\item logs y streaming
\end{itemize}

\pause

\begin{block}{Pregunta}
¿Cómo se materializan estos trade-offs en sistemas reales a escala global?
\end{block}
\end{frame}

\begin{frame}{La intuición equivocada}
No existe un sistema que maximice simultáneamente:
\begin{itemize}
\item consistencia fuerte
\item disponibilidad total
\item baja latencia global
\item simplicidad
\end{itemize}

\pause

\begin{alertblock}{Idea}
Los sistemas reales son decisiones explícitas sobre qué costo aceptar.
\end{alertblock}
\end{frame}

\UASectionSlide{CAP}

\begin{frame}{CAP correctamente leído}
CAP no significa “elegir dos de tres” como slogan.

\pause

\begin{block}{}
Si hay partición de red, un sistema replicado no puede ofrecer simultáneamente:
\begin{itemize}
\item consistencia fuerte observable
\item disponibilidad total
\end{itemize}
\end{block}
\end{frame}

\begin{frame}{Decisión bajo partición}
\begin{center}
\begin{tikzpicture}[scale=1.0,every node/.style={font=\small}]
\node (C) at (0,0) {Consistencia};
\node (A) at (3.5,0) {Disponibilidad};
\node (P) at (1.75,2.5) {Partición};

\draw[thick] (C)--(A)--(P)--(C);
\end{tikzpicture}
\end{center}

\pause

Ante partición:
\begin{itemize}
\item o rechazás operaciones para preservar corrección;
\item o respondés igual, aceptando posible inconsistencia.
\end{itemize}
\end{frame}

\begin{frame}{Qué NO dice CAP}
CAP no dice:
\begin{itemize}
\item qué sistema es “mejor”
\item que las particiones ocurran siempre
\item que consistencia y disponibilidad sean propiedades absolutas y aisladas
\end{itemize}

\pause

CAP sí dice:
\begin{itemize}
\item qué no puede garantizarse simultáneamente \emph{bajo partición}.
\end{itemize}
\end{frame}

\UASectionSlide{PACELC}

\begin{frame}{Limitación de CAP}
CAP habla cuando hay partición.

\pause

Pero incluso cuando NO hay partición, queda otra tensión:
\begin{itemize}
\item consistencia
\item latencia
\end{itemize}
\end{frame}

\begin{frame}{PACELC}
\begin{block}{}
If \textbf{P}artition: choose between \textbf{A}vailability and \textbf{C}onsistency.\\
\medskip
\textbf{E}lse: choose between \textbf{L}atency and \textbf{C}onsistency.
\end{block}
\end{frame}

\begin{frame}{Lectura práctica}
PACELC mejora CAP porque reconoce que:
\begin{itemize}
\item incluso en funcionamiento normal
\item la consistencia fuerte global
\item tiene costo en latencia
\end{itemize}
\end{frame}

\UASectionSlide{Dynamo}

\begin{frame}{Diseño de Dynamo}
Objetivo:
\begin{itemize}
\item alta disponibilidad
\item tolerancia a particiones
\item latencia baja
\end{itemize}

\pause

Elección:
\begin{itemize}
\item consistencia eventual
\item quórums flexibles
\item reconciliación posterior
\end{itemize}
\end{frame}

\begin{frame}{Componentes de Dynamo}
\begin{itemize}
\item consistent hashing
\item particionamiento
\item quórums $N, R, W$
\item versionado
\item sloppy quorum
\item hinted handoff
\item anti-entropy
\end{itemize}
\end{frame}

\begin{frame}{Quórums}
\[
R + W > N
\]

\pause

\begin{block}{Intuición}
Si lectura y escritura intersectan, aumenta probabilidad de ver datos recientes.
\end{block}

\pause

Pero:
\begin{itemize}
\item no elimina retrasos
\item no elimina conflictos
\end{itemize}
\end{frame}

\begin{frame}{Vector clocks y conflictos}
Dynamo expone versiones concurrentes y permite:
\begin{itemize}
\item detectar concurrencia
\item reconciliar después
\end{itemize}

\pause

\begin{alertblock}{Idea clave}
La disponibilidad se logra aceptando complejidad de reconciliación.
\end{alertblock}
\end{frame}

\begin{frame}{Dynamo en el espacio CAP/PACELC}
\begin{itemize}
\item Bajo partición: favorece A sobre C
\item Sin partición: favorece L sobre C
\end{itemize}

\pause

\begin{block}{Clasificación}
Dynamo es típicamente AP y PA/EL en espíritu PACELC.
\end{block}
\end{frame}

\UASectionSlide{Spanner}

\begin{frame}{Diseño de Spanner}
Objetivo:
\begin{itemize}
\item transacciones globales
\item consistencia externa
\item SQL distribuido
\end{itemize}
\end{frame}

\begin{frame}{Problema de Spanner}
¿Cómo dar una ilusión de orden global fuerte a escala planetaria?

\pause

Respuesta:
\begin{itemize}
\item consenso por shard
\item particionamiento
\item relojes acotados (TrueTime)
\end{itemize}
\end{frame}

\begin{frame}{TrueTime}
Spanner no asume reloj perfecto.

\pause

Usa un intervalo:
\[
TT.now() = [earliest, latest]
\]

\pause

Esto permite razonar sobre incertidumbre temporal de forma explícita.
\end{frame}

\begin{frame}{Commit wait}
Para asegurar orden externo, Spanner espera hasta que el commit sea seguro respecto de la incertidumbre del reloj.

\pause

\begin{block}{Costo}
Más consistencia global $\Rightarrow$ más latencia
\end{block}
\end{frame}

\begin{frame}{Spanner en CAP/PACELC}
\begin{itemize}
\item Bajo partición: preserva C, sacrifica A
\item Sin partición: sacrifica latencia para preservar C
\end{itemize}

\pause

\begin{block}{Clasificación}
Spanner es claramente CP y PC/EC en espíritu PACELC.
\end{block}
\end{frame}

\UASectionSlide{Comparación}

\begin{frame}{Dynamo vs Spanner}
\begin{center}
\begin{tabular}{lcc}
\toprule
Aspecto & Dynamo & Spanner \\
\midrule
Modelo & eventual & fuerte / externa \\
Particiones & alta tolerancia & prioriza consistencia \\
Latencia & baja & mayor \\
Conflictos & reconciliación & evita via orden fuerte \\
Caso típico & carrito, catálogo & banca, SQL global \\
\bottomrule
\end{tabular}
\end{center}
\end{frame}

\begin{frame}{Pregunta de diseño}
No se trata de “qué sistema gana”, sino de:
\begin{itemize}
\item qué dominio tenés;
\item qué errores son aceptables;
\item qué latencia podés pagar;
\item qué semántica requiere el negocio.
\end{itemize}
\end{frame}

\UASectionSlide{Cierre}

\begin{frame}{Idea central}
\begin{block}{}
La consistencia real no es una propiedad abstracta: es una decisión de arquitectura condicionada por particiones, latencia y semántica del dominio.
\end{block}
\end{frame}
